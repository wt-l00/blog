# システムコールの追加

arch/x86/entry/syscalls/syscall_64.tbl add 335 common hello sys_hello

include/linux/syscalls.h add asmlinkage long sys_hello(void)

make hello directory

in hello directory make hello.c and Makefile

hello.c

#include <linux/kernel.h>
#include <linux/sched.h>
#include <linux/cred.h>
asmlinkage long sys_hello(void)
{
    return 0xdeadbeef;
}

core-y += kernel/ mm/ fs/ ipc/ security/ crypto/ block/ hello/

再ビルドして，システムコールの動作を確かめる．

#include <unistd.h>
#include <sys/syscall.h>
#include <stdio.h>
int main() {
    printf("%lx\n", syscall(335));
    return 0;
}
